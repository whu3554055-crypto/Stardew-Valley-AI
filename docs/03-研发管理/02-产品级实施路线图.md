#### IMP-1.1.1: 数据模型设计

**文件**: `environment_system/data_models/season_config.gd`

**职责**: 定义季节配置的数据结构和验证逻辑

**实现规格**:

```gdscript
# environment_system/data_models/season_config.gd
extends Resource
class_name SeasonConfig

## 季节枚举
enum Season { SPRING, SUMMER, FALL, WINTER }

## 季节基础属性
@export var season_name: String = ""
@export var duration_days: int = 28
@export var temperature_range: Vector2 = Vector2(10, 25)
@export var rain_probability: float = 0.3

## 游戏性影响
@export var crop_growth_multiplier: float = 1.0
@export var npc_energy_regen_modifier: float = 1.0
@export var npc_mood_base_delta: float = 0.0
@export var player_energy_consumption_modifier: float = 1.0

## 视觉配置
@export var sky_color: Color = Color(0.7, 0.85, 1.0)
@export var ambient_light_energy: float = 1.0
@export var fog_density: float = 0.0
@export var particle_effect_scene: PackedScene

## 音频配置
@export var ambient_soundtrack: AudioStream
@export var weather_sound_override: Dictionary = {}

## 特殊事件列表
@export var special_events: Array[String] = []

## 验证配置有效性
func validate() -> bool:
    if season_name.is_empty():
        push_error("Season name cannot be empty")
        return false
    
    if duration_days <= 0:
        push_error("Duration must be positive")
        return false
    
    if temperature_range.x > temperature_range.y:
        push_error("Invalid temperature range")
        return false
    
    if crop_growth_multiplier < 0:
        push_error("Growth multiplier must be non-negative")
        return false
    
    return true

## 从JSON加载配置
static func from_json(data: Dictionary) -> SeasonConfig:
    var config = SeasonConfig.new()
    config.season_name = data.get("name", "")
    config.duration_days = data.get("duration_days", 28)
    config.temperature_range = Vector2(
        data.get("temp_min", 10),
        data.get("temp_max", 25)
    )
    # ... 其他字段映射
    return config

## 序列化为JSON
func to_json() -> Dictionary:
    return {
        "name": season_name,
        "duration_days": duration_days,
        "temp_min": temperature_range.x,
        "temp_max": temperature_range.y,
        "crop_growth_multiplier": crop_growth_multiplier,
        # ... 其他字段
    }
```

#### IMP-1.1.2: 季节管理器核心逻辑

**文件**: `autoload/season_manager.gd`

**职责**: 
- 管理当前季节状态
- 处理季节切换逻辑
- 广播季节事件
- 持久化季节进度

**产品级实现**:

```gdscript
# autoload/season_manager.gd
extends Node
## 季节管理器 - 单例模式
## 负责游戏内季节循环、季节效果应用、季节事件触发

signal season_will_change(old_season: String, new_season: String, days_remaining: int)
signal season_changed(old_season: String, new_season: String)
signal day_progressed(day: int, season: String, year: int)
signal season_event_triggered(event_name: String, event_data: Dictionary)

# ============================================
# 常量定义
# ============================================
const SEASON_NAMES: Array[String] = ["spring", "summer", "fall", "winter"]
const SAVE_KEY_SEASON: String = "current_season"
const SAVE_KEY_DAY: String = "current_day"
const SAVE_KEY_YEAR: String = "current_year"
const SAVE_KEY_SEASON_DAY: String = "season_day"

# ============================================
# 状态变量
# ============================================
var _current_season_index: int = 0
var _current_day: int = 1
var _current_year: int = 1
var _season_day: int = 1

var _season_configs: Dictionary = {}
var _active_effects: Array[Dictionary] = []
var _scheduled_events: Array[Dictionary] = []

# ============================================
# 生命周期
# ============================================
func _ready() -> void:
    _initialize_season_configs()
    _load_saved_state()
    _apply_current_season_effects()
    
    # 注册到游戏事件总线
    if GameManager.has_signal("day_ended"):
        GameManager.day_ended.connect(_on_day_ended)

# ============================================
# 公共API
# ============================================

## 获取当前季节名称
func get_current_season() -> String:
    return SEASON_NAMES[_current_season_index]

## 获取当前季节索引
func get_current_season_index() -> int:
    return _current_season_index

## 获取当前是第几天（总天数）
func get_current_day() -> int:
    return _current_day

## 获取当前年份
func get_current_year() -> int:
    return _current_year

## 获取当前季节的第几天
func get_season_day() -> int:
    return _season_day

## 获取当前季节配置
func get_current_season_config() -> SeasonConfig:
    return _season_configs[get_current_season()]

## 推进一天
func advance_day() -> void:
    _season_day += 1
    _current_day += 1
    
    emit_signal("day_progressed", _current_day, get_current_season(), _current_year)
    
    # 检查是否换季
    var current_config = get_current_season_config()
    if _season_day > current_config.duration_days:
        _change_to_next_season()
    else:
        # 检查是否有今日事件
        _check_daily_events()
    
    _save_state()

## 手动设置季节（用于调试或特殊事件）
func set_season(season_name: String, notify: bool = true) -> bool:
    if not SEASON_NAMES.has(season_name):
        push_error("Invalid season name: %s" % season_name)
        return false
    
    var old_index = _current_season_index
    var new_index = SEASON_NAMES.find(season_name)
    
    if old_index == new_index:
        return true  # 已经是该季节
    
    if notify:
        emit_signal("season_will_change", 
                   SEASON_NAMES[old_index], 
                   season_name, 
                   0)
    
    _remove_current_season_effects()
    _current_season_index = new_index
    _season_day = 1
    _apply_current_season_effects()
    
    if notify:
        emit_signal("season_changed", 
                   SEASON_NAMES[old_index], 
                   season_name)
    
    _save_state()
    return true

## 获取季节对某属性的修正系数
func get_season_modifier(stat_type: String) -> float:
    var config = get_current_season_config()
    
    match stat_type:
        "crop_growth":
            return config.crop_growth_multiplier
        "npc_energy_regen":
            return config.npc_energy_regen_modifier
        "npc_mood":
            return config.npc_mood_base_delta
        "player_energy":
            return config.player_energy_consumption_modifier
        _:
            push_warning("Unknown stat type: %s" % stat_type)
            return 1.0

## 注册季节事件
func register_season_event(event_name: String, trigger_condition: Callable, callback: Callable) -> void:
    """
    注册季节性事件
    
    参数:
        event_name: 事件唯一标识
        trigger_condition: 返回bool的函数，判断是否触发
        callback: 事件触发时的回调
    """
    _scheduled_events.append({
        "name": event_name,
        "condition": trigger_condition,
        "callback": callback,
        "triggered": false
    })

## 获取剩余天数到下一季节
func get_days_until_next_season() -> int:
    var config = get_current_season_config()
    return config.duration_days - _season_day

# ============================================
# 私有方法
# ============================================

func _initialize_season_configs() -> void:
    """初始化季节配置"""
    # 从配置文件加载
    var config_file = FileAccess.open("res://data/environment_configs/seasons.json", FileAccess.READ)
    if config_file:
        var json_text = config_file.get_as_text()
        config_file.close()
        
        var data = JSON.parse_string(json_text)
        if data and data.has("seasons"):
            for season_data in data.seasons:
                var config = SeasonConfig.from_json(season_data)
                if config.validate():
                    _season_configs[config.season_name] = config
                else:
                    push_error("Invalid season config: %s" % config.season_name)
    else:
        # 使用默认配置
        _load_default_seasons()

func _load_default_seasons() -> void:
    """加载默认季节配置"""
    var spring = SeasonConfig.new()
    spring.season_name = "spring"
    spring.duration_days = 28
    spring.temperature_range = Vector2(10, 25)
    spring.rain_probability = 0.3
    spring.crop_growth_multiplier = 1.0
    spring.npc_energy_regen_modifier = 1.2
    spring.npc_mood_base_delta = 0.1
    spring.sky_color = Color(0.7, 0.85, 1.0)
    _season_configs["spring"] = spring
    
    # ... 其他季节类似

func _change_to_next_season() -> void:
    """切换到下一个季节"""
    var old_season = get_current_season()
    var old_index = _current_season_index
    
    _current_season_index = (_current_season_index + 1) % SEASON_NAMES.size()
    _season_day = 1
    
    var new_season = get_current_season()
    
    # 发送即将切换信号（给UI预留动画时间）
    emit_signal("season_will_change", old_season, new_season, 0)
    
    # 移除旧季节效果
    _remove_current_season_effects()
    
    # 应用新季节效果
    _apply_current_season_effects()
    
    # 发送已切换信号
    emit_signal("season_changed", old_season, new_season)
    
    # 触发季节特殊事件
    _trigger_seasonal_events(new_season)
    
    print("[SeasonManager] Season changed: %s → %s" % [old_season, new_season])

func _apply_current_season_effects() -> void:
    """应用当前季节的全局效果"""
    var config = get_current_season_config()
    
    # 1. 更新农场生长倍率
    if has_node("/root/FarmManager"):
        get_node("/root/FarmManager").set_global_growth_multiplier(config.crop_growth_multiplier)
    
    # 2. 更新NPC行为参数
    if has_node("/root/NPCBehaviorController"):
        get_node("/root/NPCBehaviorController").update_seasonal_parameters({
            "energy_regen": config.npc_energy_regen_modifier,
            "mood_base": config.npc_mood_base_delta
        })
    
    # 3. 更新视觉效果
    _update_visual_effects(config)
    
    # 4. 更新音频
    _update_audio_effects(config)
    
    # 记录效果
    _active_effects.append({
        "type": "season",
        "season": get_current_season(),
        "applied_at": Time.get_unix_time_from_system()
    })

func _remove_current_season_effects() -> void:
    """移除当前季节的效果"""
    # 重置农场生长倍率
    if has_node("/root/FarmManager"):
        get_node("/root/FarmManager").set_global_growth_multiplier(1.0)
    
    # 清除效果记录
    _active_effects = _active_effects.filter(func(e): return e.type != "season")

func _update_visual_effects(config: SeasonConfig) -> void:
    """更新视觉特效"""
    # 天空颜色渐变
    var world_env = get_viewport().world_3d.environment
    if world_env and world_env.sky:
        var tween = create_tween()
        tween.set_parallel(true)
        tween.tween_property(world_env.sky, "sky_color", config.sky_color, 2.0)
        tween.tween_property(world_env, "ambient_light_energy", config.ambient_light_energy, 2.0)
    
    # 粒子效果切换
    if config.particle_effect_scene:
        _switch_particle_effect(config.particle_effect_scene)

func _update_audio_effects(config: SeasonConfig) -> void:
    """更新音频效果"""
    if config.ambient_soundtrack:
        AudioManager.fade_to_track(config.ambient_soundtrack, fade_duration=3.0)

func _trigger_seasonal_events(season: String) -> void:
    """触发季节特殊事件"""
    var config = get_current_season_config()
    
    for event_name in config.special_events:
        emit_signal("season_event_triggered", event_name, {
            "season": season,
            "day": _season_day,
            "year": _current_year
        })

func _check_daily_events() -> void:
    """检查并触发每日事件"""
    for event in _scheduled_events:
        if not event.triggered and event.condition.call():
            event.callback.call()
            event.triggered = true
            
            emit_signal("season_event_triggered", event.name, {})

func _on_day_ended() -> void:
    """响应GameManager的日结束信号"""
    advance_day()

func _save_state() -> void:
    """保存季节状态"""
    var save_data = {
        SAVE_KEY_SEASON: _current_season_index,
        SAVE_KEY_DAY: _current_day,
        SAVE_KEY_YEAR: _current_year,
        SAVE_KEY_SEASON_DAY: _season_day
    }
    
    GameStateManager.save_section("season", save_data)

func _load_saved_state() -> void:
    """加载保存的季节状态"""
    var save_data = GameStateManager.load_section("season")
    
    if save_data:
        _current_season_index = save_data.get(SAVE_KEY_SEASON, 0)
        _current_day = save_data.get(SAVE_KEY_DAY, 1)
        _current_year = save_data.get(SAVE_KEY_YEAR, 1)
        _season_day = save_data.get(SAVE_KEY_SEASON_DAY, 1)
    else:
        # 新游戏，使用默认值
        _current_season_index = 0
        _current_day = 1
        _current_year = 1
        _season_day = 1
```

#### IMP-1.1.3: 季节UI组件

**文件**: `scenes/ui/season_indicator.gd`

**职责**: 显示当前季节、天数、倒计时

**实现规格**:

```gdscript
# scenes/ui/season_indicator.gd
extends Control
## 季节指示器UI组件

@onready var season_label: Label = $SeasonLabel
@onready var day_label: Label = $DayLabel
@onready var countdown_label: Label = $CountdownLabel
@onready var season_icon: TextureRect = $SeasonIcon
@onready var transition_panel: Panel = $TransitionPanel
@onready var transition_label: Label = $TransitionPanel/TransitionLabel

var _icons: Dictionary = {
    "spring": preload("res://assets/ui/icons/season_spring.png"),
    "summer": preload("res://assets/ui/icons/season_summer.png"),
    "fall": preload("res://assets/ui/icons/season_fall.png"),
    "winter": preload("res://assets/ui/icons/season_winter.png")
}

func _ready() -> void:
    _connect_signals()
    _update_display()

func _connect_signals() -> void:
    SeasonManager.season_changed.connect(_on_season_changed)
    SeasonManager.day_progressed.connect(_on_day_progressed)
    SeasonManager.season_will_change.connect(_on_season_will_change)

func _update_display() -> void:
    """更新UI显示"""
    var season = SeasonManager.get_current_season()
    var day = SeasonManager.get_season_day()
    var days_left = SeasonManager.get_days_until_next_season()
    
    season_label.text = tr("SEASON_%s" % season.to_upper())
    day_label.text = tr("DAY_FORMAT").format({"day": day})
    countdown_label.text = tr("DAYS_UNTIL_NEXT_SEASON").format({"days": days_left})
    
    if _icons.has(season):
        season_icon.texture = _icons[season]

func _on_season_changed(old_season: String, new_season: String) -> void:
    """季节切换时的动画"""
    _update_display()
    
    # 播放切换动画
    var tween = create_tween()
    tween.set_ease(Tween.EASE_OUT)
    tween.set_trans(Tween.TRANS_CUBIC)
    
    tween.tween_property(self, "modulate:a", 0.0, 0.3)
    tween.tween_callback(_update_display)
    tween.tween_property(self, "modulate:a", 1.0, 0.3)

func _on_day_progressed(day: int, season: String, year: int) -> void:
    """每日推进时更新"""
    _update_display()

func _on_season_will_change(old_season: String, new_season: String, days_remaining: int) -> void:
    """季节即将切换提示"""
    if days_remaining <= 3 and days_remaining > 0:
        transition_panel.visible = true
        transition_label.text = tr("SEASON_CHANGING_SOON").format({
            "days": days_remaining,
            "new_season": tr("SEASON_%s" % new_season.to_upper())
        })
        
        # 3秒后隐藏
        await get_tree().create_timer(3.0).timeout
        transition_panel.visible = false
```

#### IMP-2.1.2: FastAPI应用入口

**文件**: `hello_agent_backend/app/main.py`

**产品级实现**:

```python
"""
Stardew AI Agent Backend Service
Production-ready FastAPI application with MCP support
"""

from contextlib import asynccontextmanager
from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import JSONResponse

from app.config import settings
from app.api.routes import mcp, npcs, world
from app.api.websocket import mcp_ws
from app.core.agent import GameAgent
from app.services.vector_store import VectorMemoryStore
from app.utils.logger import setup_logger

logger = setup_logger(__name__)


@asynccontextmanager
async def lifespan(app: FastAPI):
    """Application lifespan manager"""
    # Startup
    logger.info("Initializing Stardew AI Agent Backend...")
    
    # Initialize global agent instance
    app.state.agent = GameAgent()
    await app.state.agent.initialize()
    
    # Initialize vector store
    app.state.vector_store = VectorMemoryStore()
    await app.state.vector_store.initialize()
    
    logger.info("Backend initialized successfully")
    
    yield
    
    # Shutdown
    logger.info("Shutting down backend...")
    await app.state.agent.cleanup()
    await app.state.vector_store.cleanup()
    logger.info("Backend shutdown complete")


def create_application() -> FastAPI:
    """Create and configure FastAPI application"""
    
    application = FastAPI(
        title="Stardew AI Agent Backend",
        description="AI-powered NPC decision making and memory management for Stardew Valley clone",
        version="1.0.0",
        docs_url="/docs" if settings.DEBUG else None,
        redoc_url="/redoc" if settings.DEBUG else None,
        lifespan=lifespan
    )
    
    # CORS middleware
    application.add_middleware(
        CORSMiddleware,
        allow_origins=settings.CORS_ORIGINS,
        allow_credentials=True,
        allow_methods=["*"],
        allow_headers=["*"],
    )
    
    # Include routers
    application.include_router(mcp.router, prefix="/mcp", tags=["MCP"])
    application.include_router(npcs.router, prefix="/npcs", tags=["NPCs"])
    application.include_router(world.router, prefix="/world", tags=["World State"])
    application.include_router(mcp_ws.router, tags=["WebSocket"])
    
    # Health check endpoint
    @application.get("/health")
    async def health_check():
        return {
            "status": "healthy",
            "version": "1.0.0",
            "agent_initialized": hasattr(application.state, "agent")
        }
    
    # Global exception handler
    @application.exception_handler(Exception)
    async def global_exception_handler(request, exc):
        logger.error(f"Unhandled exception: {exc}", exc_info=True)
        return JSONResponse(
            status_code=500,
            content={"detail": "Internal server error"}
        )
    
    return application


app = create_application()


if __name__ == "__main__":
    import uvicorn
    
    uvicorn.run(
        "app.main:app",
        host=settings.HOST,
        port=settings.PORT,
        reload=settings.DEBUG,
        log_level="info"
    )
```

### Module 2.2: Agent核心逻辑

#### IMP-2.2.1: GameAgent类

**文件**: `hello_agent_backend/app/core/agent.py`

**产品级实现** (关键部分):

```python
"""
Game Agent Core
Handles NPC decision making with LLM and memory augmentation
"""

import asyncio
import json
from typing import Dict, Any, Optional, List
from datetime import datetime

from langchain_community.llms import Ollama
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import JsonOutputParser

from app.config import settings
from app.models.npc import NPCProfile, NPCDecision
from app.core.memory import MemoryManager
from app.core.tools import ToolRegistry
from app.utils.logger import get_logger

logger = get_logger(__name__)


class GameAgent:
    """
    Main agent orchestrator for NPC decision making
    
    Responsibilities:
    - Manage NPC profiles and states
    - Coordinate memory retrieval
    - Execute LLM-based decision making
    - Handle tool calls
    """
    
    def __init__(self):
        self._initialized = False
        self._npc_states: Dict[str, Dict[str, Any]] = {}
        self._memory_manager: Optional[MemoryManager] = None
        self._tool_registry: Optional[ToolRegistry] = None
        self._llm = None
        self._decision_parser = JsonOutputParser(pydantic_object=NPCDecision)
    
    async def initialize(self):
        """Initialize agent components"""
        if self._initialized:
            return
        
        logger.info("Initializing GameAgent...")
        
        # Initialize LLM
        self._initialize_llm()
        
        # Initialize memory manager
        self._memory_manager = MemoryManager()
        await self._memory_manager.initialize()
        
        # Initialize tool registry
        self._tool_registry = ToolRegistry()
        self._register_default_tools()
        
        self._initialized = True
        logger.info("GameAgent initialized")
    
    def _initialize_llm(self):
        """Initialize LLM based on configuration"""
        if settings.LLM_PROVIDER == "ollama":
            self._llm = Ollama(
                base_url=settings.OLLAMA_BASE_URL,
                model=settings.OLLAMA_MODEL,
                temperature=0.7,
                num_predict=500
            )
        elif settings.LLM_PROVIDER == "openai":
            self._llm = ChatOpenAI(
                model=settings.OPENAI_MODEL,
                temperature=0.7,
                max_tokens=500
            )
        else:
            raise ValueError(f"Unsupported LLM provider: {settings.LLM_PROVIDER}")
    
    def _register_default_tools(self):
        """Register built-in tools"""
        self._tool_registry.register_tool({
            "name": "get_npc_profile",
            "description": "Get detailed NPC profile information",
            "parameters": {"npc_id": "string"},
            "function": self._tool_get_npc_profile
        })
        
        self._tool_registry.register_tool({
            "name": "get_weather_info",
            "description": "Get current weather conditions",
            "parameters": {},
            "function": self._tool_get_weather
        })
        
        # ... more tools
    
    async def make_decision(
        self,
        npc_id: str,
        context: Dict[str, Any],
        options: Optional[List[str]] = None
    ) -> NPCDecision:
        """
        Make decision for an NPC based on context and memory
        
        Args:
            npc_id: NPC identifier
            context: Current situation context
            options: Optional list of available actions
            
        Returns:
            NPCDecision object with action and reasoning
        """
        try:
            # 1. Retrieve relevant memories
            memories = await self._memory_manager.retrieve_relevant_memories(
                npc_id=npc_id,
                query=context.get("situation", ""),
                limit=settings.MEMORY_RETRIEVAL_K
            )
            
            # 2. Get NPC profile
            npc_profile = await self._get_npc_profile(npc_id)
            
            # 3. Build prompt
            prompt = self._build_decision_prompt(
                npc_profile=npc_profile,
                context=context,
                memories=memories,
                options=options
            )
            
            # 4. Generate decision
            decision_text = await self._llm.ainvoke(prompt)
            
            # 5. Parse and validate
            decision = self._decision_parser.parse(decision_text)
            
            # 6. Store decision as memory
            await self._memory_manager.store_memory({
                "npc_id": npc_id,
                "type": "decision",
                "content": f"Decided to {decision.action}",
                "metadata": {
                    "context": context,
                    "decision": decision.dict(),
                    "timestamp": datetime.utcnow().isoformat()
                }
            })
            
            logger.info(f"NPC {npc_id} decided: {decision.action}")
            return decision
            
        except Exception as e:
            logger.error(f"Error making decision for NPC {npc_id}: {e}", exc_info=True)
            # Return fallback decision
            return NPCDecision(
                action="idle",
                reasoning=f"Error occurred: {str(e)}",
                confidence=0.0
            )
    
    def _build_decision_prompt(
        self,
        npc_profile: NPCProfile,
        context: Dict[str, Any],
        memories: List[Dict],
        options: Optional[List[str]] = None
    ) -> str:
        """Build prompt for LLM decision making"""
        
        prompt_template = """
You are embodying {npc_name} in a farming simulation game.

## Character Profile
- Occupation: {occupation}
- Personality Traits: {personality}
- Backstory: {backstory}
- Speech Style: {speech_style}

## Current Situation
Time: {time}
Season: {season}
Weather: {weather}
Location: {location}

## Recent Memories
{memories_text}

## Context
{context_description}

{options_text}

## Decision Required
Based on your character and the situation, what would you do?

Respond in JSON format:
{{
    "action": "your chosen action",
    "reasoning": "brief explanation of why",
    "confidence": 0.0-1.0,
    "emotional_state": "happy/sad/neutral/etc"
}}
"""
        
        memories_text = "\n".join([
            f"- {m['content']}" for m in memories[-3:]
        ]) if memories else "No recent memories"
        
        options_text = f"Available options: {', '.join(options)}" if options else ""
        
        return prompt_template.format(
            npc_name=npc_profile.name,
            occupation=npc_profile.occupation,
            personality=", ".join(npc_profile.personality_traits),
            backstory=npc_profile.backstory,
            speech_style=npc_profile.speech_style,
            time=context.get("time", "unknown"),
            season=context.get("season", "unknown"),
            weather=context.get("weather", "unknown"),
            location=context.get("location", "unknown"),
            memories_text=memories_text,
            context_description=context.get("situation", ""),
            options_text=options_text
        )
    
    async def cleanup(self):
        """Cleanup resources"""
        if self._memory_manager:
            await self._memory_manager.cleanup()
        self._initialized = False
```
